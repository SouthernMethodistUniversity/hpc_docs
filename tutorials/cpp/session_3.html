C++ Functors and Lambdas#

`c++ -std=c++17 -o functors_and_lamda_functions functors_and_lamda_functions.cpp

  1/*
  2 Functors and Lambda Functions
  3
  4 Useful References:
  5 * Discovering Modern C++ by Peter Gottschling (amongst many other C++ books)
  6 * https://en.cppreference.com/
  7*/
  8
  9#include <iostream> // standard IO
 10#include <vector>   // standard "vector" library for dynamic arrays
 11#include <algorithm> // for the transform functions
 12
 13/*
 14
 15A functor is essentially a class that has defined the "()" operator.
 16
 17This lets us define objects that behave like functions.
 18
 19There are 2 main reasons we might want a function like object:
 20
 21  1.  function like object allow us to define, use, and modify
 22      properties, for instance the current state of an object.
 23      In some applications this may be valueable and simplify 
 24      the code (e.g. the function "knows" informatin that you
 25      would otherwise have to pass to the function.) This also
 26      allows you to initialize frequently used values once, 
 27      which is especially useful if they're expensive to compute.   
 28
 29  2.  By using a functor, the compiler is often able to do a 
 30      better job optimizing. Essentially this happens because
 31      the compiler knows exactly what function is being called,
 32      whereas if you pass a pointer to a function, the compiler
 33      just knows a function is being called (but not necessarily
 34      which one.) This means, for example, the compiler might
 35      inline a functor.
 36
 37*/
 38
 39
 40// example of a simple functor 
 41// You see functors defined as structs instead of classes, classes are
 42// more general structure with extra features (that we're not using)
 43class increment
 44{
 45private:
 46    // define a private member variable to hold the amount we will
 47    // increment by
 48    // It is common practice to prepend member variables with "m_"
 49    // but this is purely a style choice.
 50    int m_increment_amount;
 51
 52public:
 53
 54    // constructor. This sets the default values.
 55    // This needs to be the same name as the class
 56    // Note we can overload the constructor. Here,
 57    // we're setting the increment amount to 1 if we're
 58    // not provided an argument or we're setting it
 59    // to the inputed value.
 60    increment() : m_increment_amount(1){};
 61    increment(const int &n) : m_increment_amount(n){};
 62
 63    // Now we're going to define the "()" operator 
 64    // so we can increment the input and return the
 65    // new value
 66    int operator()(const int& input_num) const
 67    {
 68        // add the increment amount to the inputed value
 69        // and return that new value
 70        return m_increment_amount + input_num;
 71    }
 72}; // end of increment class. Note, ends with ;
 73
 74
 75// function to print vectors
 76void print_vector(const std::vector<int>& v) 
 77{
 78    for (const auto& val: v) {
 79        std::cout << val << ", ";
 80    }
 81    std::cout << std::endl;
 82}
 83
 84
 85int main() {
 86
 87    // create an array of values to increment
 88    std::vector<int> v = {1, 2, 3, 4, 5}; // list initialized requires c++11 or later
 89
 90    // use our increment functor to add 1
 91    increment inc_by_one; // initialize the functor with the default 
 92    for (auto &val : v) {
 93        val = inc_by_one(val);
 94    }
 95    print_vector(v);
 96
 97    // use our increment functor to subtract 1
 98    increment dec_by_one(-1); // initialize the functor to subtract one
 99    for (auto &val : v)
100    {
101        val = inc_by_one(val);
102    }
103    print_vector(v);
104
105    // use our increment functor to add 2 to each value using the
106    // std::transform function
107    //
108    // There is a subtlety here. The increment(2) argument here is
109    // calling the constructor and setting the increment value. 
110    // Inside the transform function, it is calling v[i] = incremnt(v[i])
111    std::transform(v.begin(), v.end(), v.begin(), increment(2));
112    print_vector(v);
113
114    // subtract 2 from each entry
115    std::transform(v.begin(), v.end(), v.begin(), increment(-2));
116    print_vector(v);
117
118    // The functor we defined above is a lot of work for something that
119    // is pretty trivial (and might be even more work if the function
120    // object needed to be more complex).
121    //
122    // Since C++11, C++ has "lambda expressions," which is basically
123    // a way for the compiler to generate a functor for you using some
124    // shorthand notation
125
126    // Make a lambda function that adds 1
127    // auto requires c++14
128    // the [&] specifies we should capture values by reference
129    // we could also capture by value, pointer, etc.
130    // Then the inputs are in (...), we're taking an int
131    // Finally, we define the function in {...} (just returning input + 1)
132    auto add_one_lambda = [&](const int& val){return val + 1;};
133    std::transform(v.begin(), v.end(), v.begin(), add_one_lambda);
134    print_vector(v);
135
136    // subtract 1 using lambdas
137    auto dec_one_lambda = [&](const int &val){ return val - 1; };
138    std::transform(v.begin(), v.end(), v.begin(), dec_one_lambda);
139    print_vector(v);
140
141    // We can also use lambda functions inline. Here, we'll add 2
142    // Again, for style preferences, I find it easier to read lambda expressions
143    // if they're defined like the above. This example is inline and extactly the
144    // same, but more complicated expressions may be difficult to read.
145    std::transform(v.begin(), v.end(), v.begin(), [&](const int &val){ return val + 2; });
146    print_vector(v);
147
148    // and subtract 3
149    std::transform(v.begin(), v.end(), v.begin(), [&](const int &val){ return val - 3; });
150    print_vector(v);
151}
